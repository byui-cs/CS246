1    package macbeth;
2    
3    import java.io.BufferedReader;
4    import java.io.IOException;
5    import java.io.InputStreamReader;
6    import java.net.HttpURLConnection;
7    import java.net.URL;
8    
9    /**
10    * Helper class to send HTTP requests (the url) and receive responses.  Example use of the class:
11    *
12    * HTTPHelper http = new HTTPHelper();
13    * String result = http.readHTTP("http://api.open-notify.org/iss-now.json");
14    *
15    * Note that the URL provided can also have parameters including keys.
16    */
17   public class HTTPHelper {
18   
19       /**
20        * Will send the HTTP GET Request per the URL provided and listen for the the response.
21        *
22        * @param url - The HTTP Request URL (including any parameters)
23        * @return - The HTTP Response string.  Note that this will be null if there was an error
24        */
25       public String readHTTP(String url) {
26           try {
27               // Convert the String url to a URL object which will allow a connection to be established
28               URL urlObj = new URL(url);
29   
30               // Open a Connection to the remote site.  This will send the HTTP Get request
31               HttpURLConnection connection = (HttpURLConnection) urlObj.openConnection();
32   
33               // Open an input stream to read the response.  I prefer the BufferedReader since it provides
34               // the readLine function.
35               BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));
36   
37               // When concatenating strings read from the remote server, we get some performance improvement by
38               // using the StringBuilder class.
39               StringBuilder data = new StringBuilder();
40               String line;
41   
42               // Keep reading from the remote server until we get a null which indicates the connection has
43               // been closed.
44               do {
45                   line = reader.readLine();  // Read the next line
46                   if (line != null) {
47                       data.append(line); // Append will concatenate the new string just read into the StringBuilder
48                   }
49               }
50               while (line != null);
51   
52               // Convert the StringBuilder into a regular String for use by the calling function
53               return data.toString();
54           }
55           catch (IOException ioe) {
56               // Print out an error if something went wrong and return a null response string
57               System.out.println("Error reading HTTP Response: "+ioe);
58               return null;
59           }
60       }
61   
62   }
