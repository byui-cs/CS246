5    public class WeatherForecastSummary {
6        private String city;
7        private float maxTemp;
8        private float maxWind;
9    
10       public WeatherForecastSummary(String city, WeatherForecast forecast) {
11           this.city = city;
12           maxTemp = Float.MIN_VALUE;
13           maxWind = Float.MIN_VALUE;
14           for (WeatherForecastItem item : forecast.getForecastItems()) {
15               if (item.getMeasurements().get("temp_max") > maxTemp) {
16                   maxTemp = item.getMeasurements().get("temp_max");
17               }
18               if (item.getWind().getSpeed() > maxWind) {
19                   maxWind = item.getWind().getSpeed();
20               }
21           }
22       }
23   
24       public String getId() { return city; }
25       public float getMaxTemp() { return maxTemp; }
26       public float getMaxWind() { return maxWind; }
27   
28       public String toString() {
29           return "City: " + city +
30                   " Max Temp: " + maxTemp +
31                   " Max Wind: " + maxWind;
32       }
33   
34   }
