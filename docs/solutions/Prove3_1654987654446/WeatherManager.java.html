1    package macbeth;

3    import com.google.gson.Gson;
4    import java.util.*;
5    
6    public class WeatherManager {
7        private final String key = "YOUR_API_KEY_HERE";
8        private final String baseUrl = "https://api.openweathermap.org/data/2.5/";
9        private Gson gson;
10       private List<WeatherForecastSummary> summaries;
11       private HTTPHelper httpHelper;
12   
13       public WeatherManager() {
14           gson = new Gson();
15           httpHelper = new HTTPHelper();
16           summaries = new ArrayList<WeatherForecastSummary>();
17       }
18   
19       /* STRETCH REQUIREMENTS */
20   
21       public void loadMultipleForecasts(List<String> cityList) {
22           // The WeatherForecastSummary will have the city name and max temp and max wind from the
23           // the forecast results.
24           for (String city : cityList) {
25               WeatherForecast forecast = loadWeatherForecast(city);
26               WeatherForecastSummary summary = new WeatherForecastSummary(city, forecast);
27               summaries.add(summary);
28           }
29       }
30   
31       public void sortMaxWind() {
32           // Define what it means to sort a list of WeatherForecastSummary by Wind
33           summaries.sort(new Comparator<WeatherForecastSummary>() {
34               @Override
35               public int compare(WeatherForecastSummary o1, WeatherForecastSummary o2) {
36                   // Reuse the compare function for Floats
37                   return Float.compare(o1.getMaxWind(), o2.getMaxWind());
38               }
39           });
40       }
41   
42       public void sortMaxTemp() {
43           // Define what it means to sort a list of WeatherForecastSummary by Temp
44           // This is using a lambda instead of saying `new Comparator<WeatherForecastSummary>()`
45           summaries.sort((WeatherForecastSummary o1, WeatherForecastSummary o2)->(int)(o2.getMaxTemp()-o1.getMaxTemp()));
46       }
47   
48       public List<WeatherForecastSummary> getSummaries() { return summaries; }
49   
50       /* CORE REQUIREMENTS */
51   
52       public WeatherConditions loadCurrentWeather(String city) {
53           // Request for weather data
54           String url = baseUrl + "weather" +
55                   "?q=" + city +
56                   "&units=imperial" +
57                   "&apiKey=" + key;
58           String data = httpHelper.readHTTP(url);
59   
60           // Convert JSON data to WeatherConditions object
61           return gson.fromJson(data, WeatherConditions.class);
62       }
63   
64       public WeatherForecast loadWeatherForecast(String city) {
65           // Requset for weather forecast
66           String url = baseUrl + "forecast" +
67                   "?q=" + city +
68                   "&units=imperial" +
69                   "&apiKey=" + key;
70           String data = httpHelper.readHTTP(url);
71   
72           // Convert JSON data to WeatherForecast object
73           return gson.fromJson(data, WeatherForecast.class);
74       }

76       public static void main(String args[]) {
77           Scanner scanner = new Scanner(System.in);
78           WeatherManager mgr = new WeatherManager();
79   
80           /* CORE REQUIREMENTS */
81   
82   
83           System.out.println("Enter city name: ");
84           String city = scanner.nextLine();
85           WeatherConditions conditions = mgr.loadCurrentWeather(city);
86           System.out.println(conditions);
87           WeatherForecast forecast = mgr.loadWeatherForecast(city);
88           System.out.println(forecast);
89   
90   
91           /* STRETCH REQUIREMENTS */
92   
93           ArrayList<String> cities = new ArrayList<String>();
94           cities.add("Rexburg");
95           cities.add("Kalamazoo");
96           cities.add("Harlingen");
97           cities.add("Rockford");
98           cities.add("San Francisco");
99           System.out.println("Unsorted Summaries:");
100          mgr.loadMultipleForecasts(cities);
101          for (WeatherForecastSummary summary : mgr.getSummaries()) {
102              System.out.println(summary);
103          }
104  
105          System.out.println("Sorted by Max Temp:");
106          mgr.sortMaxTemp();
107          for (WeatherForecastSummary summary : mgr.getSummaries()) {
108              System.out.println(summary);
109          }
110  
111          System.out.println("Sorted by Max Wind:");
112          mgr.sortMaxWind();
113          for (WeatherForecastSummary summary : mgr.getSummaries()) {
114              System.out.println(summary);
115          }
116  
117      }
