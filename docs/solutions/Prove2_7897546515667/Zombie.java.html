1    package macbeth;
2    
3    import java.awt.*;
4    import java.util.*;
5    
6    /**
7     * Defines a Zombie which is a creature than can move and be aggresive.
8     *
9     * @author Chad Macbeth (CS246 Prove-02-Stretch)
10    */
11   public class Zombie extends Creature implements Movable, Aggressor {
12   
13       /**
14        * Create a Zombie
15        */
16       public Zombie() {
17           _health = 1;
18       }
19   
20       /**
21        * Zombies are squares.
22        *
23        * @return {@link Shape} of the zombie.
24        */
25       public Shape getShape() {
26           return Shape.Square;
27       }
28   
29       /**
30        * Zombies are blue.
31        *
32        * @return {$Link Color} of the zombie.
33        */
34       public Color getColor() {
35           return Color.blue;
36       }
37   
38       /**
39        * Get health status of the zombie.
40        * Zombies are alive only when their health is positive (ha ha)
41        *
42        * @return alive
43        */
44       public Boolean isAlive() {
45           return (_health > 0);
46       }
47   
48       /**
49        * Zombie will always move from left to right one square.
50        */
51       public void move() {
52           _location.x++;
53       }
54   
55       /**
56        * Inflict 10 points of health damage if the creature is not a plant.
57        *
58        * @param target The {@link Creature} we've encounterd.
59        */
60       public void attack(Creature target) {
61           // If target is null, then instanceof will return false.
62           // Therefore, need to check for null first.
63           if (target != null && !(target instanceof Plant)) {
64               target.takeDamage(10);
65           }
66       }
67   
68   }
