1    package t.macbeth.prove06;
2    
3    import androidx.appcompat.app.AppCompatActivity;
4    
5    import android.os.Bundle;
6    import android.view.View;
7    import android.widget.ArrayAdapter;
8    import android.widget.EditText;
9    import android.widget.ListView;
10   import android.widget.Toast;
11   
12   public class MainActivity extends AppCompatActivity {
13   
14       private EditText et_city;
15       private ListView lv_forecast;
16   
17       /**
18        * Initialize the activity view.
19        * 
20        * @param savedInstanceState
21        */
22       @Override
23       protected void onCreate(Bundle savedInstanceState) {
24           super.onCreate(savedInstanceState);
25           setContentView(R.layout.activity_main);
26           et_city = findViewById(R.id.et_city);
27           lv_forecast = findViewById(R.id.lv_forecast);
28       }
29   
30       /**
31        * Called when the Temp button is pressed.  Start the thread to read weather conditions
32        * to get the temp.
33        * @param view
34        */
35       public void getTemp(View view) {
36           TempTask task = new TempTask(this, et_city.getText().toString());
37           task.start();
38       }
39   
40       /**
41        * Called when the Forecast button is pressed.  Start the thread to read forecast data.
42        * @param view
43        */
44       public void getForecast(View view) {
45           ForecastTask task = new ForecastTask(this, et_city.getText().toString());
46           task.start();
47       }
48   
49       /**
50        * Called by the TempTask thread to display the temp in a toast.
51        * @param temp
52        */
53       public void displayTemp(float temp) {
54           Toast.makeText(this, "Temp is "+temp, Toast.LENGTH_SHORT).show();
55       }
56   
57       /**
58        * Called by the ForecastTask thread to display the forecast in a list view.
59        * @param forecast
60        */
61       public void displayForecast(WeatherForecast forecast) {
62           // The array adapter connects the data structure (forecast.getForecastItems()) with
63           // the ListView in the layout (lv_forecast).
64           ArrayAdapter<WeatherForecastItem> adapter = new ArrayAdapter<>(this,
65                   android.R.layout.simple_list_item_1, forecast.getForecastItems());
66           lv_forecast.setAdapter(adapter);
67       }
68   
69   }
