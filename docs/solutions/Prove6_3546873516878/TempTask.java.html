3    public class TempTask extends Thread {
4        private String city;
5        private MainActivity activity;
6        private WeatherManager manager;
7        private WeatherConditions conditions;
8    
9        /**
10        *  The activity is used in the thread to complete user interface updates.  The city
11        *  is used to load the correct weather condition data.
12        * @param activity
13        * @param city
14        */
15       public TempTask(MainActivity activity, String city) {
16           this.activity = activity;
17           this.city = city;
18           manager = new WeatherManager();
19       }
20   
21       /**
22        * Called when the start function is called on the thread.  Loads current weather
23        * data using the WeatherManager which was used on prove-03.  Instructs the
24        * activity to display the temperature at the end.
25        */
26       @Override
27       public void run() {
28           conditions = manager.loadCurrentWeather(city);
29           activity.runOnUiThread(new Runnable() {
30               @Override
31               public void run() {
32                   activity.displayTemp(conditions.getMeasurements().get("temp"));
33               }
34           });
35       }
